"LLEVA UN NOMBRE ESPECIAL POR SER EL GENERAL DE ABC" ::= PROGRAM IDENTIFICADOR COSNTANTES GLOBALES FUNCIONES  ESTRUCTURA 

ESTRUCTURA  ::= BEGIN EXPRESIONES END

/*Estructura del FOR
ESTR_FOR    ::= FOR ASIGNACION TO EXPRESION DO ESTRUCTURA

/*Estructura del IF
ESTR_IF     ::=

EXPRESIONES ::= EXPRESION | EXPRESION EXPRESIONES

EXPRESION   ::= ASIGNACION | DECLARACION | FOR | WHILE | IF 

/*OprASIGNACION es ":="
ASIGNACION  ::= DECLARAR_VARIABLES OprASIGNACION EXPRESION | IDENTIFICADOR OprASIGNACION VALOR;
VALOR       ::= INT | LONGINT | SHORTINT | CHAR | STRING | BOOLEAN | REAL

/*OprVAR es ":", VAR es la palabra que debe venir antes de definir las variables
DECLARAR_VARIABLES ::= VAR | VARIABLES 
VARIABLES          ::= VARIABLE | VARIABLE VARIABLES
VARIABLE           ::= IDENTIFICADOR OprVAR TIPO;

/*OprCONST es "=", CONST es la palabra que debe venir antes de definir las constantes
DECLARAR_CONSTANTES ::= CONST | CONSTANTES
CONSTANTES          ::= CONSTANTE | CONSTANTE CONSTANTES
CONSTANTE           ::= IDENTIFICADOR OprCONST TIPO;

TIPO        ::= INT | LONGINT | SHORTINT | CHAR | STRING | BOOLEAN | REAL
